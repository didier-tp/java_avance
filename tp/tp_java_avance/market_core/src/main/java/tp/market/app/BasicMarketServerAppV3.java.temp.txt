package tp.market.v3;

import tp.market.model.Stock;
import tp.market.persistence.dao.StockDAO;
import tp.market.persistence.dao.StockDaoJpa;
import tp.market.persistence.entity.StockEntity;
import tp.market.core.service.StockService;
import tp.market.core.service.StockServiceImpl;
import tp.market.util.MyConverter;
import tp.market.app.task.OrderExecutorTaskV1;
import tp.market.app.task.RefreshExchangesTaskV1;
import tp.market.server.task.OrderCollectorTaskJmsV2;
import tp.market.server.task.OrderCollectorTaskRmiV2;
import tp.market.server.task.OrderCollectorTaskSocketV2;

import java.util.List;

public class BasicMarketServerAppV3 {
    public static void main(String[] args) {
        System.out.println("basic market app (v3: multithread , as  server , with persistence)");
        initDataSet();

        //principale structure commune : MyStockMarket (singleton)
        //au moins 1 thread de type "collect_order"

        Thread orderCollectorThread2a = new Thread(new OrderCollectorTaskSocketV2(String.valueOf(1)));
        orderCollectorThread2a.start();

        Thread orderCollectorThread2b = new Thread(new OrderCollectorTaskRmiV2(String.valueOf(2)));
        orderCollectorThread2b.start();

        Thread orderCollectorThread2c = new Thread(new OrderCollectorTaskJmsV2(String.valueOf(3)));
        orderCollectorThread2c.start();

        //1 thread de type "execute_orders" :
        Thread orderExecutorThread = new Thread(new OrderExecutorTaskV1());
        orderExecutorThread.start();

        //1 thread de type "refresh_exchanges":
        Thread refreshExchangesThread = new Thread(new RefreshExchangesTaskV1());
        refreshExchangesThread.start();
    }

    //V3 : stock list retreive in database
    public static void initDataSet(){
        StockService stockService = new StockServiceImpl();
        StockDAO stockDAO=new StockDaoJpa();
        List<StockEntity> stockEntityList=stockDAO.findAll();
        for(StockEntity stockEntity : stockEntityList){
            Stock stock = MyConverter.map(stockEntity,Stock.class);
            stockService.addStock(stock);
        }
    }
}
